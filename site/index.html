Plugin SDK

Build a plugin in 10 lines

Write a Rust function, declare your inputs, compile to WASM. The SDK handles serialization, error propagation, and WIT bindings.

  • JSON in, JSON out ‚Äî type-safe with serde
  • Zero boilerplate via kami_tool! macro
  • Built-in helpers: parse_input, to_output
  • One-command build: cargo build --target wasm32-wasip2
  • Test locally before publishing
src/lib.rs 
use kami_guest::{abi::parse_input, abi::to_output, kami_tool};
use serde::{Deserialize, Serialize};

#[derive(Deserialize)]
struct Input { name: String }

#[derive(Serialize)]
struct Output { greeting: String }

fn handle(input: &str) -> Result<String, String> {
    let args: Input = parse_input(input)?;
    to_output(&Output {
        greeting: format!("Hello, {}!", args.name),
    })
}

kami_tool! {
    name: "dev.myorg.greet",
    version: "1.0.0",
    description: "A friendly greeting tool",
    handler: handle,
}
